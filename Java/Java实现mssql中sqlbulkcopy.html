    # 新建 BaseDao 类

    TIP

    baseDao 为常用的java数据库操作类

    @Component
public class BaseDao {

    public Connection conn = null;
    public PreparedStatement preparedStatement = null;
    public ResultSet rs = null;

    @Autowired
    private ConstantProperties constantProperties;

    /**
     * 获得连对象
     * 下面中文部分自行修改,其他代码都可以用
     * @return 返回连接对象
     */
    public Connection getConnection() {
        try {
            Class.forName("com.microsoft.sqlserver.jdbc.SQLServerDriver");
            conn = DriverManager.getConnection(
                    constantProperties.getAddonSqlUrl(), constantProperties.getAddonSqlUserName(), constantProperties.getAddonSqlPassword());
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return conn;
    }

    /**
     * 关闭所有对象
     *
     */
    public void closeAll() {
        try {
            if (rs != null) {
                rs.close();
            }
            if (preparedStatement != null) {
                preparedStatement.close();
            }
            if (conn != null) {
                conn.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    /**
     * 执行增删改操作
     *
     * @param sql 执行语句
     * @param obj 参数数组
     * @return 返回受影响的行数
     */
    public int executeUpdateSQL(String sql, Object... obj) {
        int row = 0;
        try {
            conn = getConnection();
            preparedStatement = conn.prepareStatement(sql);
            if (obj != null) {
                for (int i = 0; i < obj.length; i++) {
                    preparedStatement.setObject(i + 1, obj[i]);
                }
            }
            row = preparedStatement.executeUpdate();
        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            closeAll();
        }
        return row;
    }

    /**
     * 执行查询操作
     *
     * @param sql 查询语句
     * @param obj 参数列表
     * @return 返回对象的结果集
     */
    public ResultSet executeQuerySQL(String sql, Object... obj) {
        conn = getConnection();
        try {
            preparedStatement = conn.prepareStatement(sql);
            if (obj != null) {
                for (int i = 0; i < obj.length; i++) {
                    preparedStatement.setObject(i + 1, obj[i]);
                }
            }
            rs = preparedStatement.executeQuery();
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return rs;
    }

    # 新增 SqlBulkCopyUtils 公共方法

    WARNING

    1.getCachedRowSet 得到的默认字段必须和需要插入的字段一致

    2.批量插入前需要addColumnMapping 映射字段，这边和.NET写法不太一样

    @Component
public class SqlBulkCopyUtils {

    @Autowired
    private BaseDao baseDao;

    /**
     * 获取空的CachedRowSet对象
     * @throws SQLException
     */
    public CachedRowSetImpl getCachedRowSet() throws SQLException {
        //查询出空值用于构建CachedRowSetImpl对象以省去列映射的步骤
        ResultSet rs = baseDao.executeQuerySQL("select MobilePhone,RuleId,State,Campaign_Id,Channel from CampaignToolCouponRuleDetail where 1=0");
        CachedRowSetImpl crs = new CachedRowSetImpl();
        crs.populate(rs);
        //获取crs以后关闭数据库连接
        baseDao.closeAll();
        return crs;
    }

    /**
     * 向CachedRowSet对象插入一条数据
     * 循环调用这一个方法，将想插入数据库的数据先插入到CachedRowSet对象里
     * @param crs
     * @return
     * @throws SQLException
     */
    public CachedRowSetImpl insertIntoCachedRowSet(CachedRowSetImpl crs,AddonCampaignToolCouponRuleDetail model) throws SQLException{
        //移动指针到“插入行”，插入行是一个虚拟行
        crs.moveToInsertRow();
        //更新虚拟行的数据
        crs.updateString("MobilePhone", model.getMobilePhone());
        crs.updateString("RuleId", model.getRuleId());
        crs.updateInt("State", model.getState());
        crs.updateString("Campaign_Id", model.getCampaignId());
        crs.updateString("Channel", model.getChannel());
        //插入虚拟行
        crs.insertRow();
        //移动指针到当前行
        crs.moveToCurrentRow();
        return crs;
    }

    /**
     * 使用BulkCopy和RowSet进行批量插入
     * @param crs
     * @param batchSize
     */
    public void insertBatch(CachedRowSetImpl crs,Integer batchSize,String tableName)throws SQLException{
        SQLServerBulkCopyOptions copyOptions = new SQLServerBulkCopyOptions();
        copyOptions.setKeepIdentity(true);
        copyOptions.setBatchSize(batchSize);
        copyOptions.setUseInternalTransaction(false);
        SQLServerBulkCopy bulkCopy =
                new SQLServerBulkCopy(baseDao.getConnection());
        bulkCopy.setBulkCopyOptions(copyOptions);
        bulkCopy.addColumnMapping("MobilePhone","MobilePhone");
        bulkCopy.addColumnMapping("RuleId","RuleId");
        bulkCopy.addColumnMapping("State","State");
        bulkCopy.addColumnMapping("Campaign_Id","Campaign_Id");
        bulkCopy.addColumnMapping("Channel","Channel");
        bulkCopy.setDestinationTableName(tableName);
        bulkCopy.writeToServer(crs);
        crs.close();
        bulkCopy.close();
    }

}
